Using the ny_noaa dataset

Scatterplot

data("ny_noaa")

ny_noaa_scatter =
ny_noaa %>% 
  sample_n(10000) %>% 
  separate(date, into = c("year", "month", "day")) %>% 
  mutate(
    year = as.integer(year),
    month = as.integer(month),
    day = as.integer(day),
    tmax = as.integer(tmax),
    tmin = as.integer(tmin),
    prcp = prcp / 10,
    tmax = tmax / 10,
    tmin = tmin / 10,
    prcp = as.integer(prcp),
    tmax = as.integer(tmax),
    tmin = as.integer(tmin),
  ) %>% 
  group_by(id, year, month) %>% 
  summarize(
    mean_tmax = mean(tmax, na.rm = TRUE)
  ) %>%
  filter(month %in% c(01, 07))
## `summarise()` regrouping output by 'id', 'year' (override with `.groups` argument)
ny_noaa_scatter %>% 
  mutate(
    text_label = str_c("Station Id: ", id, "\nYear: ", year, "\nMonth: ", month)
  ) %>% 
  plot_ly(
    x = ~year, y = ~mean_tmax, text = ~text_label, color = ~month,
    alpha = .5, type = "scatter", mode = "markers"
  )

Bar plot

ny_noaa_bar =
ny_noaa %>% 
  sample_n(10000) %>% 
  separate(date, into = c("year", "month", "day")) %>% 
  mutate(
    year = as.integer(year),
    month = as.integer(month),
    day = as.integer(day),
    tmax = as.integer(tmax),
    tmin = as.integer(tmin),
    prcp = prcp / 10,
    tmax = tmax / 10,
    tmin = tmin / 10,
    prcp = as.integer(prcp),
    tmax = as.integer(tmax),
    tmin = as.integer(tmin),
  ) %>% 
  group_by(year, month) %>%
  arrange(year) %>% 
  summarize(
    mean_tmax = mean(tmax, na.rm = TRUE)
  ) %>% 
  filter(mean_tmax != "NaN")
## `summarise()` regrouping output by 'year' (override with `.groups` argument)
ny_noaa_bar %>%
  filter(month %in% c(07)) %>% 
  plot_ly(
    x = ~year, y = ~mean_tmax, color = ~mean_tmax,
    alpha = .6, type = "bar"
  )
